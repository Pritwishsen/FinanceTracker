2. ADD processRecurringSchedules() TO DataService (called on every app load):
   static async processRecurringSchedules() {
     const schedules = this.getRecurring();
     const today = new Date().toISOString().split("T")[0];
     let changed = false;
     for (const schedule of schedules) {
       if (!schedule.isActive) continue;
       if (schedule.endDate && today > schedule.endDate) continue;
       while (schedule.nextDue <= today) {
         const instance = { ...schedule.templateExpense,
           id: Date.now() + Math.random(),
           date: schedule.nextDue,
           createdAt: new Date().toISOString() };
         if (schedule.type === "expense") await this.addExpense(instance);
         else await this.addIncome(instance);
         schedule.lastGenerated = schedule.nextDue;
         schedule.nextDue = computeNextDue(schedule.nextDue, schedule.frequency);
         changed = true;
       }
     }
     if (changed) this.saveRecurring(schedules);
   }
